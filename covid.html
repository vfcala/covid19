# dataset with confirmed cases worldwide by date and Country 
confirmed_cases_worldwide = cases_worldwide %>% filter(type=="confirmed") %>% select(date, cases) %>% group_by(date) %>% summarize(n=sum(cases))
death_cases_worldwide = cases_worldwide %>% filter(type=="death") %>% select(date, cases) %>% group_by(date) %>% summarize(n=sum(cases))

# dataframe with cumulative confirmed and death cases per day 
confirmed_cases_worldwide = confirmed_cases_worldwide %>% mutate(cum_confirmed_cases = cumsum(n)) %>% select(date, cum_confirmed_cases)
death_cases_worldwide = death_cases_worldwide %>% mutate(cum_death_cases=cumsum(n)) %>% select(date, cum_death_cases)
df = cbind(confirmed_cases_worldwide, death_cases_worldwide)
df = df[,-3]
df = df %>% mutate(ratio=cum_death_cases/cum_confirmed_cases)

# draw line plot to visualize the confirmed cases worldwide
g1 = ggplot(df, aes(x=date, y=cum_confirmed_cases)) +
  geom_line() +
  ylab("Cumulative confirmed cases")
g1

# draw line plot to visualize the death cases worldwide 
g2 = ggplot(df, aes(x=date, y=cum_death_cases)) +
  geom_line() +
  ylab("Cumulative death cases")
g2

# draw line plot of the ratio death/confirmed worldwide -> overall not constant 
g3 = ggplot(df, aes(x=date, y=ratio)) + 
  geom_line() + 
  ylab("Ratio between death and confirmed cases")
g3

# dataframe with cumulative confirmed cases divided by China vs rest of the world
china = cases_worldwide %>% filter(Country.Region=='China') %>% select(date, cases, type)
world = cases_worldwide %>% filter(Country.Region!='China') %>% select(date, cases, type)

confirmed_china = china %>% 
  filter(type=='confirmed') %>% 
  select(date, cases) %>% 
  group_by(date) %>% 
  summarize(n=sum(cases)) %>% 
  mutate(cum_cases=cumsum(n)) %>% 
  select(date, cum_cases) %>% 
  mutate(is_china=T)

confirmed_world = world %>% 
  filter(type=='confirmed') %>% 
  select(date, cases) %>% 
  group_by(date) %>% 
  summarize(n=sum(cases)) %>% 
  mutate(cum_cases=cumsum(n)) %>% 
  select(date, cum_cases) %>%
  mutate(is_china=F)
  

df2 = rbind(confirmed_china, confirmed_world)

# draw line plots to compare confirmed cases in China vs the rest of the World
g4 = ggplot(data=df2) + 
  geom_line(aes(x=date, y=cum_cases, col=is_china)) +
  ylab("Cumulative confirmed cases")
g4

# zoom before the intersection 
df3 = df2 %>% filter(date <= '2020-03-17')
g5 = ggplot(data=df3) +
  geom_line(aes(x=date, y=cum_cases, col=is_china)) + 
  ylab("Cumulative confirmed cases")
g5

# what happened? keep trace of important events
events <- tribble(
  ~ date, ~ event,
  "2020-01-30", "Global health\nemergency declared",
  "2020-03-11", "Pandemic\ndeclared",
  "2020-02-13", "China reporting\nchange"
) %>%
  mutate(date = as.Date(date))

# add who_events to the previous graphic as vertical dashed lines
g5 +
  geom_vline(data=events, mapping=aes(xintercept=date), linetype='dashed') +
  geom_text(data=events, mapping=aes(x=date,label=event), y=100000)

# add trend lines (method=linear model) starting from 15th of February (when China's reportings change)
df4 = df2 %>% filter(date >= '2020-02-15' & date <= '2020-03-17') %>% filter(is_china==T)
df5 = df2 %>% filter(date >= '2020-02-15' & date <= '2020-03-17') %>% filter(is_china==F)

# China's confirmed cases growth rate is slower than linear
ggplot(data=df4, aes(x=date, y=cum_cases)) + 
  geom_line() +
  ylab("Cumulative confirmed cases") + 
  geom_smooth(se=F, method='lm')
## `geom_smooth()` using formula 'y ~ x'

# Rest of the World's confirmed cases growth rate is exponential (log scale)
ggplot(data=df5, aes(x=date, y=cum_cases)) + 
  geom_line() +
  ylab("Cumulative confirmed cases") + 
  geom_smooth(se=F, method='lm') +
  scale_y_log10()
## `geom_smooth()` using formula 'y ~ x'

# top 5 most infected Countries 
top5 = cases_worldwide %>% filter(type=='confirmed') %>% select(Country.Region, date, cases) %>%
  group_by(Country.Region) %>% mutate(cum_cases=cumsum(cases)) %>% group_by(Country.Region) %>% 
  summarize(total=max(cum_cases)) %>% top_n(n=5, wt=total)

df_top5 = cases_worldwide %>% filter(type=='confirmed') %>% select(Country.Region, date, cases) %>% 
  filter(Country.Region %in% c("China", "Germany", "Italy", "Spain", "US")) %>% group_by(date, Country.Region) %>% summarize(n=sum(cases)) %>%
  group_by(Country.Region) %>% mutate(cum_cases=cumsum(n))

# line plots of top 5 most infected Countries
ggplot(data=df_top5, aes(x=date, y=cum_cases, group=Country.Region, col=Country.Region)) +
